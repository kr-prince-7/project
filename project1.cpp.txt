void upmove(int a[4][4]){
    int i,j,li,ri;
    for(j=0;j<4;j++){
        li=0,ri=j;
        for(i=0;i<4;i++){
            if(a[i][j]!=0){
                if(a[i-1][j]==0||a[i-1][j]==a[i][j]){
                    if(a[li][ri]==a[i][j]){
                        a[li][ri]*=2;
                        a[i][j]=0;
                    }
                    else
                    {
						if(a[li][ri]==0)
						{
							a[li][ri]=a[i][j];
							a[i][j]=0;
						}
						else
						{
							a[++li][ri]=a[i][j];
							a[i][j]=0;
						}
					}
                }
                else li++;
            }

        }
    }
}



void down(int a[4][4])
{
	int i,j,li,ri;
	for(j=0;j<4;j++)
	{
		li=3,ri=j;
		for(i=2;i>=0;i--)
		{
			if(a[i][j]!=0)
			{
				if(a[i+1][j]==0 || a[i+1][j]==a[i][j])
				{
					if(a[li][ri]==a[i][j])
					{
						a[li][ri]*=2;
						a[i][j]=0;
					}
					else
					{
						if(a[li][ri]==0)
						{
							a[li][ri]=a[i][j];
							a[i][j]=0;
						}
						else
						{
							a[--li][ri]=a[i][j];
							a[i][j]=0;
						}
					}
				}
				else li--;
	}	}	}	}



void left(int a[4][4])
{
	int i,j,li,ri;
	for(i=0;i<4;i++)

		li=i,ri=0;
		for(j=1;j<4;j++)
